    const trialBalance = [];
    let totalDebits = 0;
    let totalCredits = 0;

    for (const account of accounts) {
      // Calculate balance for each account from journal entries
      const balance = await db.select({
        totalDebits: sum(journalEntryLines.debitAmount),
        totalCredits: sum(journalEntryLines.creditAmount)
      })
      .from(journalEntryLines)
      .where(eq(journalEntryLines.accountId, account.id))
      .groupBy(journalEntryLines.accountId);

      const debits = parseFloat(balance[0]?.totalDebits || '0');
      const credits = parseFloat(balance[0]?.totalCredits || '0');
      const netBalance = debits - credits;

      // Determine if balance should be debit or credit based on account type
      let debitBalance = 0;
      let creditBalance = 0;

      if (['Asset', 'Expense'].includes(account.accountType)) {
        // Assets and Expenses have debit balances
        if (netBalance > 0) {
          debitBalance = netBalance;
          totalDebits += netBalance;
        } else {
          creditBalance = Math.abs(netBalance);
          totalCredits += Math.abs(netBalance);
        }
      } else {
        // Liabilities, Equity, Revenue have credit balances
        if (netBalance < 0) {
          creditBalance = Math.abs(netBalance);
          totalCredits += Math.abs(netBalance);
        } else {
          debitBalance = netBalance;
          totalDebits += netBalance;
        }
      }

      if (debitBalance > 0 || creditBalance > 0) {
        trialBalance.push({
          accountCode: account.accountCode,

          creditBalance: creditBalance.toFixed(2)
        });
